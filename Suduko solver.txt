#include <iostream>
using namespace std;
int grid[9][9];
bool isValid(int r, int c, int num) {
    for (int i = 0; i < 9; i++) {
        if (grid[r][i] == num || grid[i][c] == num ||
            grid[r - r % 3 + i / 3][c - c % 3 + i % 3] == num)
            return false;
    }
    return true;
}
bool solve() {
    for (int r = 0; r < 9; r++)
        for (int c = 0; c < 9; c++)
            if (grid[r][c] == 0) {
                for (int num = 1; num <= 9; num++) {
                    if (isValid(r, c, num)) {
                        grid[r][c] = num;
                        if (solve()) return true;
                        grid[r][c] = 0;
                    }
                }
                return false;
            }
    return true;
}
void printGrid() {
    for (int r = 0; r < 9; r++) {
        for (int c = 0; c < 9; c++)
            cout << grid[r][c] << " ";
        cout << "\n";
    }
}
int main() {
    cout << "Enter Sudoku (0 for empty cells):\n";
    for (int r = 0; r < 9; r++)
        for (int c = 0; c < 9; c++)
            cin >> grid[r][c];
    if (solve()) {
        cout << "\nSolved Sudoku:\n";
        printGrid();
    } else {
        cout << "No solution exists.\n";
    }
    return 0;
}
